  def do_rank
    @hp = Hpricot(open("http://www.google.com.br/search?num=100&hl=en&query=unipharmus"))
    pesquisa = @hp.search('li')
    @results = []  
    pesquisa.each do |p|
      @result = p.search('a[@href*=www.unipharmus.com.br]')
      unless @result.empty?
        @results << p.position + 1
      end
    end
    return @results
  end
